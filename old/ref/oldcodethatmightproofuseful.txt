; === Logging Functions ===
LogMessage(level, message) {
    timestamp := FormatTime(, "yyyy-MM-dd HH:mm:ss")
    logEntry := timestamp " [" level "] " message "`n"

    try {
        FileAppend(logEntry, LogFile)
    } catch Error as e {
        ; If we can't write to the log file, fall back to OutputDebug
        OutputDebug("Log Error: " e.Message " - Original message: " logEntry)
    }

    ; For errors, also show a tray notification
    if (level = LOG_ERROR) {
        TrayTip("Error: " message, "SAP Automation Error", 0x3)  ; 0x3 = Error icon
    }
}

; === SAP Functions ===
ConnectSAP() {
    WinWaitActive("ahk_exe" "saplogon.exe")
    try {
        ; Initialize SAP connection
        global SapGui := ComObjGet("SAPGUI").GetScriptingEngine
        global SapSession := SapGui.ActiveSession
        LogMessage(LOG_INFO, "Successfully connected to SAP")
        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Failed to connect to SAP: " e.Message)
        decision := MsgBox("Could not connect to SAP window. Retry?`n" e.Message, "Connection Error", 0x5)
        if (decision == "Retry")
            return ConnectSAP()
        else {
            LogMessage(LOG_ERROR, "User chose not to retry SAP connection. Exiting.")
            ExitApp()
        }
    }
}

; Enhanced Function to find and set a field value by name with better subscreen support
FindSapField(fieldName, dynpro := "") {
    if (!IsSet(SapSession)) {
        LogMessage(LOG_ERROR, "SAP session not available for field search")
        return false
    }

    try {
        ; Log the search attempt
        LogMessage(LOG_DEBUG, "Searching for SAP field: " fieldName (dynpro ? " (Dynpro: " dynpro ")" : ""))

        ; Try to find the field directly first
        try {
            field := SapSession.findById("wnd[0]/usr/" fieldName)
            LogMessage(LOG_INFO, "Found field directly: " fieldName)
            return field
        } catch Error as e {
            LogMessage(LOG_DEBUG, "Direct field search failed, trying alternative methods")
        }

        ; If dynpro is specified, try to use it
        if (dynpro != "") {
            try {
                field := SapSession.findById("wnd[0]/usr/dynpro=" dynpro "/" fieldName)
                LogMessage(LOG_INFO, "Found field with dynpro: " fieldName)
                return field
            } catch Error as e {
                LogMessage(LOG_DEBUG, "Dynpro field search failed")
            }
        }

        ; Try common prefixes if direct search failed
        prefixes := ["txt", "ctxt", "btn", "cmb", "rad", "chk", "lbl", "tab"]
        for prefix in prefixes {
            try {
                field := SapSession.findById("wnd[0]/usr/" prefix fieldName)
                LogMessage(LOG_INFO, "Found field with prefix " prefix ": " fieldName)
                return field
            } catch Error as e {
                ; Continue to next prefix
            }
        }

        ; Try to search within subscreens - this is the key addition
        try {
            userArea := SapSession.findById("wnd[0]/usr")
            field := SearchFieldRecursive(userArea, fieldName, "wnd[0]/usr")
            if (field) {
                LogMessage(LOG_INFO, "Found field via recursive search: " fieldName)
                return field
            }
        } catch Error as e {
            LogMessage(LOG_DEBUG, "Recursive search failed: " e.Message)
        }

        LogMessage(LOG_WARNING, "Could not find field: " fieldName)
        return false
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error searching for field " fieldName ": " e.Message)
        return false
    }
}

; Helper function to recursively search for a field in SAP controls
SearchFieldRecursive(parentControl, targetFieldName, currentPath := "") {
    if (!IsObject(parentControl)) {
        return false
    }

    try {
        ; Check if the current control matches what we're looking for
        controlId := ""
        try {
            controlId := parentControl.Id
        } catch {
            ; No ID available
        }

        ; Check if this control's ID contains our target field name
        if (InStr(controlId, targetFieldName)) {
            return parentControl
        }

        ; Try to access children
        try {
            children := parentControl.Children
            if (IsObject(children)) {
                childCount := children.Count

                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childId := child.Id
                        childPath := currentPath "/" childId

                        ; Check if this child is what we're looking for
                        if (InStr(childId, targetFieldName)) {
                            LogMessage(LOG_DEBUG, "Found field at path: " childPath)
                            return child
                        }

                        ; Recursively search this child's children
                        result := SearchFieldRecursive(child, targetFieldName, childPath)
                        if (result) {
                            return result
                        }
                    } catch Error as e {
                        ; Skip this child
                    }
                }
            }
        } catch Error as e {
            ; No children or can't access them
        }
    } catch Error as e {
        ; Error with this control
    }

    return false
}

            field.Text := value

        }

        LogMessage(LOG_ERROR, "Error setting field via direct path: " e.Message)

    }
}
; Function to set a field value by name
SetSapField(fieldName, value, dynpro := "") {
    field := FindSapField(fieldName, dynpro)
    if (!field) {
        LogMessage(LOG_ERROR, "Could not find field to set: " fieldName)
        return false
    }

    try {
        ; Try to set the value based on field type
        fieldType := field.Type
        LogMessage(LOG_DEBUG, "Setting field " fieldName " of type " fieldType " to value: " value)

        if (fieldType = "GuiTextField" || fieldType = "GuiCTextField" || fieldType = "GuiPasswordField") {
            field.Text := value
            LogMessage(LOG_INFO, "Set text field " fieldName " to: " value)
            return true
        }
        else if (fieldType = "GuiComboBox") {
            field.Key := value  ; For combo box, set by key
            LogMessage(LOG_INFO, "Set combo box " fieldName " to key: " value)
            return true
        }
        else if (fieldType = "GuiCheckBox") {
            if (value = true || value = 1 || value = "X")
                field.Selected := true
            else
                field.Selected := false
            LogMessage(LOG_INFO, "Set checkbox " fieldName " to: " (field.Selected ? "checked" : "unchecked"))
            return true
        }
        else if (fieldType = "GuiRadioButton") {
            field.Selected := true
            LogMessage(LOG_INFO, "Selected radio button: " fieldName)
            return true
        }
        else if (fieldType = "GuiButton") {
            field.Press()
            LogMessage(LOG_INFO, "Pressed button: " fieldName)
            return true
        }
        else {
            ; For other field types, try generic Text property
            field.Text := value
            LogMessage(LOG_INFO, "Set field " fieldName " to: " value)
            return true
        }
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error setting field " fieldName ": " e.Message)
        return false
    }
}

; Function to get a field value by name
GetSapField(fieldName, dynpro := "") {
    field := FindSapField(fieldName, dynpro)
    if (!field) {
        LogMessage(LOG_ERROR, "Could not find field to get: " fieldName)
        return ""
    }

    try {
        ; Try to get the value based on field type
        fieldType := field.Type

        if (fieldType = "GuiTextField" || fieldType = "GuiCTextField" || fieldType = "GuiPasswordField") {
            value := field.Text
            LogMessage(LOG_DEBUG, "Got text field " fieldName " value: " value)
            return value
        }
        else if (fieldType = "GuiComboBox") {
            value := field.Key  ; For combo box, get the key
            LogMessage(LOG_DEBUG, "Got combo box " fieldName " key: " value)
            return value
        }
        else if (fieldType = "GuiCheckBox") {
            value := field.Selected
            LogMessage(LOG_DEBUG, "Got checkbox " fieldName " value: " (value ? "checked" : "unchecked"))
            return value ? "X" : ""
        }
        else if (fieldType = "GuiRadioButton") {
            value := field.Selected
            LogMessage(LOG_DEBUG, "Got radio button " fieldName " value: " (value ? "selected" : "not selected"))
            return value ? "X" : ""
        }
        else {
            ; For other field types, try generic Text property
            try {
                value := field.Text
                LogMessage(LOG_DEBUG, "Got field " fieldName " value: " value)
                return value
            } catch {
                LogMessage(LOG_WARNING, "Could not get text value from field: " fieldName)
                return ""
            }
        }
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error getting field " fieldName " value: " e.Message)
        return ""
    }
}

CreateDropHandle(fileNames, x := 0, y := 0) {
    fileNames := RegExReplace(fileNames, "\n$")
    fileNames := RegExReplace(fileNames, "^\n")
    dropHandle := DllCall("GlobalAlloc", "UInt", 0x42, "UPtr", 20 + StrLen(fileNames) + 2)
    pMemory := DllCall("GlobalLock", "UPtr", dropHandle)
    NumPut("UPtr", 20, pMemory + 0)   ; offset
    NumPut("UPtr", x, pMemory + 4)    ; pt.x
    NumPut("UPtr", y, pMemory + 8)    ; pt.y
    NumPut("UPtr", 0, pMemory + 12)   ; fNC
    NumPut("UPtr", 0, pMemory + 16)   ; fWide
    pDestination := pMemory + 20
    Loop Parse, fileNames, "`n", "`r" {
        DllCall("RtlMoveMemory", "UPtr", pDestination, "AStr", A_LoopField, "UPtr", StrLen(A_LoopField))
        pDestination += StrLen(A_LoopField) + 1
    }
    DllCall("GlobalUnlock", "UPtr", dropHandle)
    Return dropHandle
}

; === Excel Functions ===
ExcelOpen(filePath) {
    try {
        if (!FileExist(filePath)) {
            LogMessage(LOG_ERROR, "Excel file does not exist: " filePath)
            TrayTip("Excel file not found: " filePath, "File Error", 0x3)
            return false
        }

        LogMessage(LOG_INFO, "Opening Excel file: " filePath)
        ; Try to create the Excel application
        global ExcelApp := ComObject("Excel.Application")
        if !IsSet(ExcelApp) {
            LogMessage(LOG_ERROR, "Failed to create Excel application")
            TrayTip("Failed to create Excel application", "Excel Error", 0x3)
            return false
        }

        ExcelApp.Visible := true

        ; Try to open the workbook
        global ExcelWorkbook := ExcelApp.Workbooks.Open(filePath)
        if !IsSet(ExcelWorkbook) {
            LogMessage(LOG_ERROR, "Failed to open workbook")
            TrayTip("Failed to open workbook", "Excel Error", 0x3)
            ExcelApp.Quit()
            ExcelApp := ""
            return false
        }

        ; Try to access the worksheet
        global ExcelWorksheet := ExcelWorkbook.Worksheets(1)
        if !IsSet(ExcelWorksheet) {
            LogMessage(LOG_ERROR, "Failed to access worksheet")
            TrayTip("Failed to access worksheet", "Excel Error", 0x3)
            ExcelWorkbook.Close(false)  ; Don't save
            ExcelApp.Quit()
            ExcelWorkbook := ""
            ExcelApp := ""
            return false
        }

        global ExcelIsOpen := true
        LogMessage(LOG_INFO, "Excel file opened successfully")
        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error opening Excel file: " e.Message)
        TrayTip("Error opening Excel file. Check log for details.", "Excel Error", 0x3)
        CleanupExcel()  ; Attempt to clean up
        return false
    }
}

        return false
    }
}

ExcelReadCell(row, col) {
    if (!ExcelIsOpen || !IsSet(ExcelWorksheet)) {
        LogMessage(LOG_WARNING, "Attempted to read Excel cell but worksheet is not open")
        return ""
    }

    try {
        value := ExcelWorksheet.Cells(row, col).Value
        LogMessage(LOG_DEBUG, "Read from cell [" row "," col "]: " value)
        return value
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error reading cell [" row "," col "]: " e.Message)
        return ""
    }
}

ExcelWriteCell(row, col, value) {
    if (!ExcelIsOpen || !IsSet(ExcelWorksheet)) {
        LogMessage(LOG_WARNING, "Attempted to write Excel cell but worksheet is not open")
        return false
    }

    try {
        LogMessage(LOG_DEBUG, "Writing to cell [" row "," col "]: " value)
        ExcelWorksheet.Cells(row, col).Value := value
        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error writing to cell [" row "," col "]: " e.Message)
        return false
    }
}

ExcelSelectSheet(sheetNameOrIndex) {
    if (!ExcelIsOpen || !IsSet(ExcelWorkbook)) {
        LogMessage(LOG_WARNING, "Attempted to select sheet but workbook is not open")
        return false
    }

    try {
        LogMessage(LOG_INFO, "Selecting worksheet: " sheetNameOrIndex)
        global ExcelWorksheet := ExcelWorkbook.Worksheets(sheetNameOrIndex)
        ExcelWorksheet.Activate()
        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error selecting worksheet: " e.Message)
        return false
    }
}

ExcelGetLastRow(column := "A") {
    if (!ExcelIsOpen || !IsSet(ExcelWorksheet)) {
        LogMessage(LOG_WARNING, "Attempted to get last row but worksheet is not open")
        return 0
    }
    ;loop until column cell is empty, then return row number
    ;this is to prevent functions formats etc from counting as row.
    try {
        lastRow := 1
        while (ExcelWorksheet.Cells(lastRow, column).Value != "") {
            lastRow++
        }
        LogMessage(LOG_INFO, "Last row in column " column ": " (lastRow - 1))
        return lastRow - 1  ; Return the last filled row
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error finding last row: " e.Message)
        return 0
    }
}

ExcelGetLastColumn(row := 1) {
    if (!ExcelIsOpen || !IsSet(ExcelWorksheet)) {
        LogMessage(LOG_WARNING, "Attempted to get last column but worksheet is not open")
        return 0
    }

    try {
        lastCol := ExcelWorksheet.Cells(row, ExcelWorksheet.Columns.Count).End(-4159).Column
        LogMessage(LOG_INFO, "Last column in row " row ": " lastCol)
        return lastCol
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error finding last column: " e.Message)
        return 0
    }
}

ExcelReadRange(startRow, startCol, endRow, endCol) {
    if (!ExcelIsOpen || !IsSet(ExcelWorksheet)) {
        LogMessage(LOG_WARNING, "Attempted to read range but worksheet is not open")
        return []
    }

    try {
        LogMessage(LOG_DEBUG, "Reading range [" startRow "," startCol "] to [" endRow "," endCol "]")
        range := ExcelWorksheet.Range(ExcelWorksheet.Cells(startRow, startCol), ExcelWorksheet.Cells(endRow, endCol))
        data := range.Value
        return data
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error reading range: " e.Message)
        return []
    }
}

; Enhanced Excel closing function with better error handling
ExcelClose(saveChanges := false) {
    if (!ExcelIsOpen) {
        LogMessage(LOG_INFO, "Excel close requested but Excel is not open")
        return true  ; Already closed or never opened
    }

    try {
        LogMessage(LOG_INFO, "Closing Excel" (saveChanges ? " with save" : " without save"))

        ; Save workbook if requested and if workbook exists
        if (saveChanges && IsSet(ExcelWorkbook)) {
            try {
                ExcelWorkbook.Save()
                LogMessage(LOG_INFO, "Excel file saved")
            } catch Error as e {
                LogMessage(LOG_WARNING, "Could not save Excel file: " e.Message)
                TrayTip("Could not save Excel file", "Excel Warning", 0x2)
            }
        }

        ; Close the workbook if it exists
        if (IsSet(ExcelWorkbook)) {
            try {
                ExcelWorkbook.Close(false)  ; Don't save again
                LogMessage(LOG_INFO, "Excel workbook closed")
            } catch Error as e {
                LogMessage(LOG_WARNING, "Could not close workbook properly: " e.Message)
            }
        }

        ; Quit Excel if it exists
        if (IsSet(ExcelApp)) {
            try {
                ExcelApp.Quit()
                LogMessage(LOG_INFO, "Excel application quit")
            } catch Error as e {
                LogMessage(LOG_WARNING, "Could not quit Excel properly: " e.Message)
            }
        }

        ; Clean up COM objects regardless of any errors
        global ExcelWorksheet := ""
        global ExcelWorkbook := ""
        global ExcelApp := ""
        global ExcelIsOpen := false

        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error closing Excel: " e.Message)
        TrayTip("Error closing Excel. Check log for details.", "Excel Error", 0x3)

        ; Clean up anyway
        global ExcelWorksheet := ""
        global ExcelWorkbook := ""
        global ExcelApp := ""
        global ExcelIsOpen := false

        return false
    }
}

; Additional function to ensure Excel is closed properly
CleanupExcel() {
    if (!ExcelIsOpen) {
        return
    }

    LogMessage(LOG_INFO, "Performing Excel cleanup")

    ; Try to clean up Excel even if there are errors
    try {
        if (IsSet(ExcelWorkbook)) {
            try {
                ExcelWorkbook.Close(false)
                LogMessage(LOG_INFO, "Excel workbook closed during cleanup")
            } catch Error as e {
                LogMessage(LOG_WARNING, "Cleanup error closing workbook: " e.Message)
            }
        }

        if (IsSet(ExcelApp)) {
            try {
                ExcelApp.Quit()
                LogMessage(LOG_INFO, "Excel application quit during cleanup")
            } catch Error as e {
                LogMessage(LOG_WARNING, "Cleanup error quitting Excel: " e.Message)
            }
        }
    } catch Error as e {
        LogMessage(LOG_WARNING, "Excel cleanup error: " e.Message)
    }

    ; Clear variables
    global ExcelWorksheet := ""
    global ExcelWorkbook := ""
    global ExcelApp := ""
    global ExcelIsOpen := false
}
; Function to dump all SAP screen elements to a file for debugging
DumpSapScreen(outputFileName := "") {
    if (!IsSet(SapSession)) {
        LogMessage(LOG_ERROR, "SAP session not available for screen dump")
        return false
    }

    ; Generate default filename if not provided
    if (outputFileName = "") {
        outputFileName := A_ScriptDir "\logs\SAPScreenDump_" FormatTime(, "yyyy-MM-dd_HH-mm-ss") ".txt"
    }

    try {
        ; Create or clear the output file
        try {
            FileDelete(outputFileName)
        } catch Error as e {
            ; File doesn't exist, that's fine
        }

        FileAppend("=== SAP SCREEN DUMP ===`n", outputFileName)
        FileAppend("Date: " FormatTime(, "yyyy-MM-dd HH:mm:ss") "`n", outputFileName)
        FileAppend("User: " A_UserName "`n", outputFileName)

        ; Get current screen info
        try {
            titleText := SapSession.findById("wnd[0]/titl").Text
            FileAppend("Screen Title: " titleText "`n", outputFileName)
        } catch Error as e {
            FileAppend("Screen Title: [Could not retrieve]`n", outputFileName)
        }

        try {
            programInfo := SapSession.Info.Program
            transactionCode := SapSession.Info.Transaction
            FileAppend("Program: " programInfo "`n", outputFileName)
            FileAppend("Transaction: " transactionCode "`n", outputFileName)
        } catch Error as e {
            FileAppend("Program/Transaction: [Could not retrieve]`n", outputFileName)
        }

        FileAppend("`n=== SCREEN ELEMENTS ===`n`n", outputFileName)

        ; Start recursive scan from the main window
        try {
            mainWindow := SapSession.findById("wnd[0]")
            ScanSapControl(mainWindow, "wnd[0]", outputFileName, 0)
            LogMessage(LOG_INFO, "SAP screen dump created: " outputFileName)
            TrayTip("Screen dump created: " outputFileName, "Debug Info", 0x1)
            return true
        } catch Error as e {
            LogMessage(LOG_ERROR, "Error starting SAP screen dump: " e.Message)
            FileAppend("ERROR: " e.Message, outputFileName)
            return false
        }
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error creating SAP screen dump: " e.Message)
        return false
    }
}

; Helper function for recursive scanning of SAP controls
ScanSapControl(control, id, outputFile, depth := 0) {
    if (!IsObject(control)) {
        return
    }

    try {
        ; Create indent based on depth
        indent := ""
        Loop depth {
            indent .= "    "  ; 4 spaces per level
        }

        ; Get control properties
        controlType := "[Unknown]"
        try {
            controlType := control.Type
        } catch Error as e {
            ; Type property not available
        }

        ; Write basic control info
        FileAppend(indent "ID: " id "`n", outputFile)
        FileAppend(indent "Type: " controlType "`n", outputFile)

        ; Try to get common properties based on control type
        if (controlType = "GuiTextField" || controlType = "GuiCTextField" || controlType = "GuiPasswordField") {
            try {
                FileAppend(indent "Text: " control.Text "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }

            try {
                FileAppend(indent "Name: " control.Name "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }
        }
        else if (controlType = "GuiComboBox") {
            try {
                FileAppend(indent "Key: " control.Key "`n", outputFile)
                FileAppend(indent "Text: " control.Text "`n", outputFile)
            } catch Error as e {
                ; Properties not available
            }

            ; Try to get all entries in the combo box
            try {
                entries := []
                entriesCollection := control.Entries
                entryCount := entriesCollection.Count

                Loop entryCount {
                    try {
                        entry := entriesCollection.Item(A_Index - 1)  ; COM collections are often 0-based
                        entries.Push(entry.Key " - " entry.Text)
                    } catch Error as e {
                        ; Skip this entry
                    }
                }

                if (entries.Length > 0) {
                    FileAppend(indent "Entries:`n", outputFile)
                    for entryText in entries {
                        FileAppend(indent "    - " entryText "`n", outputFile)
                    }
                }
            } catch Error as e {
                ; Can't get entries
            }
        }
        else if (controlType = "GuiCheckBox" || controlType = "GuiRadioButton") {
            try {
                FileAppend(indent "Selected: " (control.Selected ? "Yes" : "No") "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }
        }
        else if (controlType = "GuiButton") {
            try {
                FileAppend(indent "Text: " control.Text "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }

            try {
                FileAppend(indent "Tooltip: " control.Tooltip "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }
        }
        else if (controlType = "GuiLabel") {
            try {
                FileAppend(indent "Text: " control.Text "`n", outputFile)
            } catch Error as e {
                ; Property not available
            }
        }
        else if (controlType = "GuiTab") {
            try {
                FileAppend(indent "Text: " control.Text "`n", outputFile)
                FileAppend(indent "Selected: " (control.Selected ? "Yes" : "No") "`n", outputFile)
            } catch Error as e {
                ; Properties not available
            }
        }
        else if (controlType = "GuiTableControl") {
            try {
                rowCount := control.RowCount
                colCount := control.ColumnCount
                FileAppend(indent "Rows: " rowCount "`n", outputFile)
                FileAppend(indent "Columns: " colCount "`n", outputFile)

                ; Optionally dump table headers
                try {
                    FileAppend(indent "Column Headers:`n", outputFile)
                    Loop colCount {
                        colName := control.GetColumnHeader(A_Index - 1)
                        FileAppend(indent "    - Column " (A_Index - 1) ": " colName "`n", outputFile)
                    }
                } catch Error as e {
                    ; Can't get column headers
                }
            } catch Error as e {
                ; Properties not available
            }
        }
        else if (controlType = "GuiTree" || (controlType = "GuiShell" && InStr(id, "shell"))) {
            try {
                nodeCount := control.NodeCount
                FileAppend(indent "Node Count: " nodeCount "`n", outputFile)

                if (nodeCount > 0 && nodeCount < 100) {  ; Limit to prevent huge dumps
                    FileAppend(indent "Nodes:`n", outputFile)
                    Loop nodeCount {
                        try {
                            nodeName := control.GetNodeKeyByIndex(A_Index - 1)  ; SAP uses 0-based indexing
                            FileAppend(indent "    - " nodeName "`n", outputFile)
                        } catch Error as e {
                            ; Skip this node
                        }
                    }
                }
            } catch Error as e {
                ; Can't get node information
            }
        }

        ; Add more properties that might be useful
        try {
            ; Try to get additional properties
            if HasProp(control, "Visible") {
                FileAppend(indent "Visible: " (control.Visible ? "Yes" : "No") "`n", outputFile)
            }
            if HasProp(control, "Enabled") {
                FileAppend(indent "Enabled: " (control.Enabled ? "Yes" : "No") "`n", outputFile)
            }
            if HasProp(control, "Changeable") {
                FileAppend(indent "Changeable: " (control.Changeable ? "Yes" : "No") "`n", outputFile)
            }
        } catch Error as e {
            ; Properties not available
        }

        ; Add a separator line
        FileAppend(indent "---------------`n", outputFile)

        ; Try to get children - not all controls have a Children property
        try {
            children := control.Children

            if (IsObject(children)) {
                childCount := children.Count
                FileAppend(indent "Children (" childCount "):`n", outputFile)

                ; Process each child
                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childId := id "/" child.Id
                        FileAppend("`n", outputFile)
                        ScanSapControl(child, childId, outputFile, depth + 1)
                    } catch Error as e {
                        FileAppend(indent "    [Error getting child " (A_Index - 1) ": " e.Message "]`n", outputFile)
                    }
                }
            }
        } catch Error as e {
            ; No children or can't access them
        }

    } catch Error as e {
        FileAppend(indent "[Error scanning control: " e.Message "]`n", outputFile)
    }
}

; Helper function to list fields recursively
ListFieldsRecursive(control, path, outputFile) {
    if (!IsObject(control)) {
        return
    }

    try {
        ; Check if this is an input field
        controlType := ""
        try {
            controlType := control.Type
        } catch Error as e {
            ; Type not available
        }

        ; Only list interactive controls
        if (controlType = "GuiTextField" || controlType = "GuiCTextField" ||
            controlType = "GuiPasswordField" || controlType = "GuiComboBox" ||
            controlType = "GuiCheckBox" || controlType = "GuiRadioButton" ||
            controlType = "GuiButton") {

            fieldInfo := path " - " controlType

            try {
                if HasProp(control, "Name") {
                    fieldInfo .= " - Name: " control.Name
                }
            } catch Error as e {
                ; Name not available
            }

            try {
                if HasProp(control, "Text") {
                    fieldInfo .= " - Text: " control.Text
                }
            } catch Error as e {
                ; Text not available
            }

            FileAppend(fieldInfo "`n", outputFile)
        }

        ; Process children
        try {
            children := control.Children
            if (IsObject(children)) {
                childCount := children.Count
                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childPath := path "/" child.Id
                        ListFieldsRecursive(child, childPath, outputFile)
                    } catch Error as e {
                        ; Skip this child
                    }
                }
            }
        } catch Error as e {
            ; No children or can't access them
        }
    } catch Error as e {
        ; Skip this control
    }
}
; Function to handle SAP message popup and extract text
HandleSapMessagePopup(rowIndex) {
    if (!IsSet(SapSession)) {
        LogMessage(LOG_ERROR, "SAP session not available for popup handling")
        return ""
    }

    try {
        ; Check if a popup window exists (usually wnd[1] for popups)
        popup := SapSession.findById("wnd[1]")

        ; Get popup type and title
        popupTitle := ""
        try {
            popupTitle := popup.Text
            LogMessage(LOG_INFO, "Found popup: " popupTitle)
        } catch {
            LogMessage(LOG_DEBUG, "Could not get popup title")
        }

        ; Try to extract message text from common popup locations
        messageText := ""

        ; Try different common paths for message text in popups
        messagePaths := [
            "wnd[1]/usr/txtMESSTXT1",
            "wnd[1]/usr/txtMESSAGE",
            "wnd[1]/usr/txt*",
            "wnd[1]/usr/lbl*"
        ]

        for path in messagePaths {
            try {
                msgField := SapSession.findById(path)
                messageText := msgField.Text
                if (messageText != "") {
                    LogMessage(LOG_INFO, "Extracted message: " messageText)
                    break
                }
            } catch {
                ; Try next path
            }
        }

        ; If still no message, try to get all text from the popup
        if (messageText = "") {
            try {
                ; Get user area and scan for text
                userArea := popup.findById("usr")
                messageText := ScanPopupForText(userArea)
            } catch {
                LogMessage(LOG_DEBUG, "Could not scan popup user area")
            }
        }

        ; Write to Excel Sheet 6
        if (messageText != "") {
            WriteMessageToExcel(rowIndex, messageText, popupTitle)
        }

        ; Close the popup (usually with Enter or OK button)
        try {
            ; Try OK button first
            okButton := popup.findById("wnd[1]/tbar[0]/btn[0]")
            okButton.Press()
            LogMessage(LOG_INFO, "Closed popup with OK button")
        } catch {
            ; If no OK button, try Enter key
            try {
                popup.sendVKey(0)  ; Enter key
                LogMessage(LOG_INFO, "Closed popup with Enter key")
            } catch {
                LogMessage(LOG_WARNING, "Could not close popup automatically")
            }
        }

        return messageText

    } catch Error as e {
        LogMessage(LOG_DEBUG, "No popup found or error handling popup: " e.Message)
        return ""
    }
}

; Helper function to scan popup for text
ScanPopupForText(control) {
    allText := ""

    if (!IsObject(control)) {
        return ""
    }

    try {
        ; Check if this control has text
        controlType := control.Type

        if (controlType = "GuiTextField" || controlType = "GuiLabel" || controlType = "GuiTitlebar") {
            try {
                text := control.Text
                if (text != "" && !InStr(text, "___")) {  ; Exclude separator lines
                    allText .= text . " "
                }
            } catch {
                ; No text available
            }
        }

        ; Check children
        try {
            children := control.Children
            if (IsObject(children)) {
                childCount := children.Count
                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childText := ScanPopupForText(child)
                        if (childText != "") {
                            allText .= childText . " "
                        }
                    } catch {
                        ; Skip this child
                    }
                }
            }
        } catch {
            ; No children
        }
    } catch {
        ; Skip this control
    }

    return Trim(allText)
}

; Function to write message to Excel Sheet 6
WriteMessageToExcel(rowIndex, messageText, popupTitle := "") {
    if (!ExcelIsOpen || !IsSet(ExcelWorkbook)) {
        LogMessage(LOG_WARNING, "Cannot write to Excel - workbook not open")
        return false
    }

    try {
        ; Switch to Sheet 6
        if (!ExcelSelectSheet(6)) {
            ; Create Sheet 6 if it doesn't exist
            try {
                global ExcelWorksheet := ExcelWorkbook.Worksheets.Add()
                ExcelWorksheet.Name := "SAP Messages"

                ; Add headers
                ExcelWriteCell(1, 1, "Row Index")
                ExcelWriteCell(1, 2, "Timestamp")
                ExcelWriteCell(1, 3, "Message Type")
                ExcelWriteCell(1, 4, "Message Text")
                ExcelWriteCell(1, 5, "Vendor Name")

                LogMessage(LOG_INFO, "Created Sheet 6 for SAP messages")
            } catch Error as e {
                LogMessage(LOG_ERROR, "Could not create Sheet 6: " e.Message)
                return false
            }
        }

        ; Find next empty row in Sheet 6
        lastMessageRow := ExcelGetLastRow("A") + 1

        ; Write message data
        ExcelWriteCell(lastMessageRow, 1, rowIndex)
        ExcelWriteCell(lastMessageRow, 2, FormatTime(, "yyyy-MM-dd HH:mm:ss"))
        ExcelWriteCell(lastMessageRow, 3, popupTitle)
        ExcelWriteCell(lastMessageRow, 4, messageText)

        ; Get vendor name from Sheet 1 for reference
        ExcelSelectSheet(1)
        vendorName := ExcelReadCell(rowIndex, 3)  ; Column C = NAME
        ExcelSelectSheet(6)
        ExcelWriteCell(lastMessageRow, 5, vendorName)

        LogMessage(LOG_INFO, "Wrote SAP message to Sheet 6, row " lastMessageRow)

        ; Switch back to Sheet 1
        ExcelSelectSheet(1)

        return true
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error writing message to Excel: " e.Message)
        return false
    }
}
; Add this function around line 850 (after other dump functions)

    }

                }

                return false
            }
        } else {
            ; Try to access popup directly
            popup := SapSession.findById("wnd[1]")
        }
        
        ; Clear/create the output file
        try {
            FileDelete(outputFile)
        } catch {
            ; File doesn't exist, that's fine
        }
        
        ; Write header
        FileAppend("=== SAP POPUP FIELD DUMP ===`n", outputFile)
        FileAppend("Date: " FormatTime(, "yyyy-MM-dd HH:mm:ss") "`n", outputFile)
        FileAppend("User: " A_UserName "`n", outputFile)
        FileAppend("Generated by: anonymusdennis`n`n", outputFile)
        
        ; Get popup title
        popupTitle := ""
        try {
            popupTitle := popup.Text
            FileAppend("Popup Title: " popupTitle "`n", outputFile)
        } catch {
            FileAppend("Popup Title: [Could not retrieve]`n", outputFile)
        }
        
        ; Get popup type
        popupType := ""
        try {
            popupType := popup.Type
            FileAppend("Popup Type: " popupType "`n", outputFile)
        } catch {
            FileAppend("Popup Type: [Could not retrieve]`n", outputFile)
        }
        
        FileAppend("`n=== INTERACTIVE FIELDS ONLY ===`n", outputFile)
        FileAppend("(Fields you can fill/interact with)`n`n", outputFile)
        
        ; Scan for interactive fields only
        interactiveFields := []
        ScanForInteractiveFields(popup, "wnd[1]", &interactiveFields)  ; Pass by reference
        
        ; Sort and write interactive fields
        if (interactiveFields.Length > 0) {
            for field in interactiveFields {
                FileAppend(field "`n", outputFile)
            }
        } else {
            FileAppend("[No interactive fields found]`n", outputFile)
        }
        
        FileAppend("`n=== ALL CONTROLS (DETAILED) ===`n`n", outputFile)
        
        ; Do a complete scan
        ScanPopupControl(popup, "wnd[1]", outputFile, 0)
        
        ; Log success
        LogMessage(LOG_INFO, "Popup field dump created: " outputFile)
        
        ; Show notification with file location
        TrayTip("Popup fields dumped!`nCheck: " outputFile, "Success", 0x1)
        
        ; Also create a simplified field list
        CreateSimplifiedFieldList(outputFile)
        
        return true
        
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error dumping popup fields: " e.Message)
        FileAppend("`nERROR: " e.Message "`n", outputFile)
        return false
    }
}

; Helper function to scan specifically for interactive fields
ScanForInteractiveFields(control, path, &fieldArray) {  ; fieldArray is already a reference parameter
    if (!IsObject(control)) {
        return
    }
    
    try {
        ; Check if this is an interactive field
        controlType := ""
        try {
            controlType := control.Type
        } catch {
            ; Type not available
        }
        
        ; Interactive field types
        interactiveTypes := ["GuiTextField", "GuiCTextField", "GuiPasswordField", 
                           "GuiComboBox", "GuiCheckBox", "GuiRadioButton", "GuiButton"]
        
        isInteractive := false
        for iType in interactiveTypes {
            if (controlType = iType) {
                isInteractive := true
                break
            }
        }
        
        if (isInteractive) {
            fieldInfo := "Path: " path "`n"
            fieldInfo .= "Type: " controlType "`n"
            
            ; Get additional properties
            try {
                fieldInfo .= "Name: " control.Name "`n"
            } catch {
                ; No name
            }
            
            try {
                if (controlType = "GuiTextField" || controlType = "GuiCTextField") {
                    fieldInfo .= "Current Value: " control.Text "`n"
                } else if (controlType = "GuiComboBox") {
                    fieldInfo .= "Current Key: " control.Key "`n"
                    fieldInfo .= "Current Text: " control.Text "`n"
                }
            } catch {
                ; No value
            }
            
            try {
                fieldInfo .= "Changeable: " (control.Changeable ? "Yes" : "No") "`n"
            } catch {
                ; Property not available
            }
            
            fieldInfo .= "---`n"
            fieldArray.Push(fieldInfo)
        }
        
        ; Scan children
        try {
            children := control.Children
            if (IsObject(children)) {
                childCount := children.Count
                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childPath := path "/" child.Id
                        ScanForInteractiveFields(child, childPath, &fieldArray)  ; Pass reference
                    } catch {
                        ; Skip this child
                    }
                }
            }
        } catch {
            ; No children
        }
    } catch {
        ; Skip this control
    }
}
; Enhanced popup control scanner
ScanPopupControl(control, id, outputFile, depth := 0) {
    if (!IsObject(control)) {
        return
    }
    
    try {
        ; Create indent
        indent := ""
        Loop depth {
            indent .= "  "
        }
        
        ; Get control type
        controlType := "[Unknown]"
        try {
            controlType := control.Type
        } catch {
            ; Type not available
        }
        
        ; Write control info
        FileAppend(indent "ID: " id "`n", outputFile)
        FileAppend(indent "Type: " controlType "`n", outputFile)
        
        ; Get all available properties based on type
        if (InStr(controlType, "Field") || InStr(controlType, "Combo")) {
            try {
                FileAppend(indent "Text: " control.Text "`n", outputFile)
            } catch {
                ; Property not available
            }
            
            try {
                FileAppend(indent "Name: " control.Name "`n", outputFile)
            } catch {
                ; Property not available
            }
            
            try {
                FileAppend(indent "Tooltip: " control.Tooltip "`n", outputFile)
            } catch {
                ; Property not available
            }
            
            try {
                FileAppend(indent "MaxLength: " control.MaxLength "`n", outputFile)
            } catch {
                ; Property not available
            }
        }
        
        ; Check standard properties
        try {
            FileAppend(indent "Visible: " (control.Visible ? "Yes" : "No") "`n", outputFile)
        } catch {
            ; Property not available
        }
        
        try {
            FileAppend(indent "Changeable: " (control.Changeable ? "Yes" : "No") "`n", outputFile)
        } catch {
            ; Property not available
        }
        
        FileAppend(indent "---`n", outputFile)
        
        ; Process children
        try {
            children := control.Children
            if (IsObject(children)) {
                childCount := children.Count
                Loop childCount {
                    try {
                        child := children.ElementAt(A_Index - 1)
                        childId := id "/" child.Id
                        ScanPopupControl(child, childId, outputFile, depth + 1)
                    } catch {
                        ; Skip this child
                    }
                }
            }
        } catch {
            ; No children
        }
    } catch {
        FileAppend(indent "[Error scanning control]`n", outputFile)
    }
}

; Function to check and log any SAP messages (statusbar or popup)
CheckAndLogSapMessages(context := "") {
    if (!IsSet(SapSession)) {
        LogMessage(LOG_ERROR, "SAP session not available for message checking")
        return ""
    }
    
    messageFound := false
    allMessages := ""
    
    ; First check for popup messages (higher priority)
    try {
        popup := SapSession.findById("wnd[1]")
        popupTitle := ""
        popupMessage := ""
        
        try {
            popupTitle := popup.Text
        } catch {
            popupTitle := "[Unknown Popup]"
        }
        
        ; Try to extract message text
        try {
            ; Try common message field paths
            messagePaths := [
                "wnd[1]/usr/txtMESSTXT1",
                "wnd[1]/usr/txtMESSAGE",
                "wnd[1]/usr/lbl[1]",
                "wnd[1]/usr/lbl[2]"
            ]
            
            for path in messagePaths {
                try {
                    msgField := SapSession.findById(path)
                    text := msgField.Text
                    if (text != "" && !InStr(text, "___")) {
                        popupMessage .= text . " "
                    }
                } catch {
                    ; Continue to next path
                }
            }
            
            ; If no message found, scan the popup
            if (popupMessage = "") {
                userArea := popup.findById("usr")
                popupMessage := ScanPopupForText(userArea)
            }
        } catch {
            ; Could not extract message
        }
        
        if (popupMessage != "") {
            LogMessage(LOG_INFO, "POPUP MESSAGE" (context ? " (" context ")" : "") ": " popupTitle " - " popupMessage)
            allMessages .= "POPUP: " popupMessage . "`n"
            messageFound := true
            
            ; Try to close the popup
            try {
                popup.findById("tbar[0]/btn[0]").Press()  ; OK button
            } catch {
                try {
                    popup.sendVKey(0)  ; Enter key
                } catch {
                    ; Could not close automatically
                }
            }
        }
    } catch {
        ; No popup found
    }
    
    ; Check statusbar for messages
    try {
        statusBar := SapSession.findById("wnd[0]/sbar")
        statusText := statusBar.Text
        statusMessageType := statusBar.MessageType
        
        if (statusText != "") {
            ; Determine message severity
            messageSeverity := "INFO"
            if (statusMessageType = "E") {
                messageSeverity := "ERROR"
            } else if (statusMessageType = "W") {
                messageSeverity := "WARNING"
            } else if (statusMessageType = "S") {
                messageSeverity := "SUCCESS"
            } else if (statusMessageType = "I") {
                messageSeverity := "INFO"
            } else if (statusMessageType = "A") {
                messageSeverity := "ABORT"
            }
            
            LogMessage(LOG_INFO, "STATUSBAR " messageSeverity (context ? " (" context ")" : "") ": " statusText)
            allMessages .= "STATUSBAR [" messageSeverity "]: " statusText . "`n"
            messageFound := true
        }
    } catch Error as e {
        LogMessage(LOG_DEBUG, "Could not check statusbar: " e.Message)
    }
    
    return Trim(allMessages)
}

; Enhanced function to monitor messages continuously
MonitorSapMessages(duration := 5000, checkInterval := 500) {
    if (!IsSet(SapSession)) {
        return ""
    }
    
    startTime := A_TickCount
    collectedMessages := ""
    lastStatusMessage := ""
    
    while (A_TickCount - startTime < duration) {
        ; Check statusbar
        try {
            statusBar := SapSession.findById("wnd[0]/sbar")
            currentStatus := statusBar.Text
            
            if (currentStatus != "" && currentStatus != lastStatusMessage) {
                messageType := statusBar.MessageType
                timestamp := FormatTime(, "HH:mm:ss")
                
                LogMessage(LOG_INFO, "SAP Message at " timestamp " [" messageType "]: " currentStatus)
                collectedMessages .= timestamp " [" messageType "]: " currentStatus . "`n"
                lastStatusMessage := currentStatus
            }
        } catch {
            ; Statusbar not accessible
        }
        
        ; Check for popups
        try {
            popup := SapSession.findById("wnd[1]")
            ; Popup found, handle it
            CheckAndLogSapMessages("Monitor detected popup")
            break  ; Exit loop if popup found
        } catch {
            ; No popup
        }
        
        Sleep checkInterval
    }
    
    return collectedMessages
}

; Function to get just the current message without closing popups
GetCurrentSapMessage() {
    if (!IsSet(SapSession)) {
        return ""
    }
    
    ; Check popup first
    try {
        popup := SapSession.findById("wnd[1]")
        userArea := popup.findById("usr")
        popupText := ScanPopupForText(userArea)
        if (popupText != "") {
            return "POPUP: " popupText
        }
    } catch {
        ; No popup
    }
    
    ; Check statusbar
    try {
        statusBar := SapSession.findById("wnd[0]/sbar")
        statusText := statusBar.Text
        if (statusText != "") {
            return "STATUSBAR: " statusText
        }
    } catch {
        ; No statusbar message
    }
    
 
    return ""
}
; Function to handle document attachment popup
HandleDocumentAttachment(rowIndex) {
    if (!IsSet(SapSession)) {
        LogMessage(LOG_ERROR, "SAP session not available for document attachment")
        return false
    }
    
    try {
        ; Read Sheet 4 data
        if (!ExcelSelectSheet(4)) {
            LogMessage(LOG_WARNING, "Sheet 3 not found, skipping document attachment")
            return false
        }
        
        ; Read document data
        DocData_structured := DocData_structured_empty
        DocData_structured.DOKUMENTART := ExcelReadCell(rowIndex, 1)  ; Column A
        DocData_structured.PFAD := ExcelReadCell(rowIndex, 2)  ; Column B
        DocData_structured.KOMMENTAR := ExcelReadCell(rowIndex, 3)  ; Column C
        
        ; Check if we have document data
        if (DocData_structured.DOKUMENTART = "" || DocData_structured.PFAD = "") {
            LogMessage(LOG_INFO, "No document data for row " rowIndex)
            ExcelSelectSheet(1)  ; Switch back to Sheet 1
            return false
        }
        
        ; Handle relative paths - if path starts with . replace with script directory
        originalPath := DocData_structured.PFAD
        if (SubStr(originalPath, 1, 1) = ".") {
            ; Remove the leading . and combine with script directory
            relativePath := SubStr(originalPath, 2)
            originalPath := A_ScriptDir . relativePath
            LogMessage(LOG_DEBUG, "Converted relative path to: " originalPath)
        }
        
        ; Check if source file exists
        if (!FileExist(originalPath)) {
            LogMessage(LOG_ERROR, "Source file does not exist: " originalPath)
            ExcelSelectSheet(1)
            return false
        }
        
        ; Create SAP temp directory path dynamically for current user
        sapTempDir := "C:\Users\" A_UserName "\AppData\Local\SAP\SAP GUI\tmp\"
        
        ; Create directory if it doesn't exist
        if (!DirExist(sapTempDir)) {
            try {
                DirCreate(sapTempDir)
                LogMessage(LOG_INFO, "Created SAP temp directory: " sapTempDir)
            } catch Error as e {
                LogMessage(LOG_ERROR, "Could not create SAP temp directory: " e.Message)
                ExcelSelectSheet(1)
                return false
            }
        }
        
        ; Define target path with fixed filename
        targetPath := sapTempDir "upload.xlsx"
        
        ; Copy file to SAP temp directory
        try {
            FileCopy(originalPath, targetPath, true)  ; true = overwrite if exists
            LogMessage(LOG_INFO, "Copied file from: " originalPath " to: " targetPath)
        } catch Error as e {
            LogMessage(LOG_ERROR, "Failed to copy file: " e.Message)
            ExcelSelectSheet(1)
            return false
        }
        
        LogMessage(LOG_INFO, "Attaching document for row " rowIndex)
        LogMessage(LOG_DEBUG, "Document type: " DocData_structured.DOKUMENTART)
        LogMessage(LOG_DEBUG, "Using temp path: " targetPath)
        LogMessage(LOG_DEBUG, "Comment: " DocData_structured.KOMMENTAR)
        
        ; Wait for popup to appear
        Sleep SleepTimeMedium
        
        ; Fill popup fields with the temp file path
        if (DocData_structured.DOKUMENTART != "")
            SetSapFieldDirect("wnd[1]/usr/ctxt/BUP/AQ0VBDOKUME-DOKART", DocData_structured.DOKUMENTART)
        
        ; Use the temp path instead of original
        SetSapFieldDirect("wnd[1]/usr/ctxt/BUP/AQ0VBDOKUME-DNAME", targetPath)
        
        if (DocData_structured.KOMMENTAR != "")
            SetSapFieldDirect("wnd[1]/usr/txt/BUP/AQ0VBDOKUME-DTEXT", DocData_structured.KOMMENTAR)
        
        ; Confirm the popup using the OK button
        try {
            ; Press OK button
            SapSession.findById("wnd[1]/usr/btnBTN_OK").Press()
            LogMessage(LOG_INFO, "Document attachment confirmed with OK button")
            
            ; Optional: Delete temp file after successful upload
            Sleep SleepTimeMedium  ; Give SAP time to process
            try {
                FileDelete(targetPath)
                LogMessage(LOG_DEBUG, "Deleted temp file: " targetPath)
            } catch {
                LogMessage(LOG_DEBUG, "Could not delete temp file (may still be in use)")
            }
            
        } catch Error as e {
            LogMessage(LOG_ERROR, "Could not confirm document attachment: " e.Message)
            
            ; Try to clean up temp file on error
            try {
                FileDelete(targetPath)
            } catch {
                ; Ignore cleanup errors
            }
            
            ExcelSelectSheet(1)
            return false
        }
        
        ExcelSelectSheet(1)  ; Switch back to Sheet 1
        return true
        
    } catch Error as e {
        LogMessage(LOG_ERROR, "Error in document attachment: " e.Message)
        ExcelSelectSheet(1)  ; Switch back to Sheet 1
        return false
    }
}
